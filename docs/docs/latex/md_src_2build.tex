 \href{https://github.com/mrsinho/shvulkan}{\texttt{ }}\hypertarget{md_src_2build_autotoc_md10}{}\doxysection{\texorpdfstring{Clone and Build}{Clone and Build}}\label{md_src_2build_autotoc_md10}
Open the terminal and run the following commands\+:

 

Tested on Win32 and X11.


\begin{DoxyCode}{0}
\DoxyCodeLine{git\ clone\ -\/-\/recursive\ https://github.com/mrsinho/shvulkan.git}
\DoxyCodeLine{cd\ shvulkan}
\DoxyCodeLine{mkdir\ build}
\DoxyCodeLine{cd\ build}
\DoxyCodeLine{cmake\ -\/DSH\_VULKAN\_BUILD\_EXAMPLES=ON\ ..}
\DoxyCodeLine{cmake\ -\/-\/build\ .}

\end{DoxyCode}




Tested on X11 and Wayland graphics servers. The Nix flake lets the system choose the default windowing system.

\begin{DoxyWarning}{Warning}
While testing with Wayland revealed no issues, there are some resizing issues when using X11.
\end{DoxyWarning}

\begin{DoxyCode}{0}
\DoxyCodeLine{git\ clone\ -\/-\/recursive\ https://github.com/mrsinho/shvulkan.git}
\DoxyCodeLine{cd\ shvulkan}
\DoxyCodeLine{nix\ build}
\DoxyCodeLine{nix\ develop}
\DoxyCodeLine{cd\ result/examples/bin}
\DoxyCodeLine{./shvulkan-\/clear-\/color\ \&\ ./shvulkan-\/scene\ \&\ ./shvulkan-\/compute-\/power-\/numbers}

\end{DoxyCode}

To link to the {\ttfamily shvulkan} library with CMake\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{if\ (NOT\ TARGET\ shvulkan)}
\DoxyCodeLine{}
\DoxyCodeLine{set(SH\_VULKAN\_ROOT\_DIR\ \ \ \ \ path/to/shvulkan/root/directory)}
\DoxyCodeLine{set(SH\_VULKAN\_BINARIES\_DIR\ path/your/binaries/directory)}
\DoxyCodeLine{}
\DoxyCodeLine{include(\$\{SH\_VULKAN\_ROOT\_DIR\}/shvulkan/shvulkan.cmake)}
\DoxyCodeLine{build\_shvulkan()}
\DoxyCodeLine{}
\DoxyCodeLine{endif()}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ [...]}
\DoxyCodeLine{}
\DoxyCodeLine{target\_link\_libraries(app\ PUBLIC\ shvulkan)}

\end{DoxyCode}
